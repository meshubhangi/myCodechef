for t in range(int(input())):
    s=[]
    d=[]
    for i in range(int(input())):
        s.append(input())
        d.append(input())
    for l in s:
        if l not in d:
            p=l
    i=1
    while i==1:
        n=s.index(p)
        print(p+"-"+d[n], end=' ')
        if d[n] in s:
            p=d[n]
        else:
            i=0
